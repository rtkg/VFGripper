182  virtual int spinOnce(){
183 
184  /* restart if timed out */
185  unsigned long c_time = hardware_.time();
186  if( (c_time - last_sync_receive_time) > (SYNC_SECONDS*2200) ){
187  configured_ = false;
188  }
189 
190  /* reset if message has timed out */
191  if ( mode_ != MODE_FIRST_FF){
192  if (c_time > last_msg_timeout_time){
193  mode_ = MODE_FIRST_FF;
194  }
195  }
196 
197  /* while available buffer, read data */
198  while( true )
199  {
200  int data = hardware_.read();
201  if( data < 0 )
202  break;
203  checksum_ += data;
204  if( mode_ == MODE_MESSAGE ){ /* message data being recieved */
205  message_in[index_++] = data;
206  bytes_--;
207  if(bytes_ == 0) /* is message complete? if so, checksum */
208  mode_ = MODE_MSG_CHECKSUM;
209  }else if( mode_ == MODE_FIRST_FF ){
210  if(data == 0xff){
211  mode_++;
212  last_msg_timeout_time = c_time + MSG_TIMEOUT;
213  }
214  else if( hardware_.time() - c_time > (SYNC_SECONDS)){
215  /* We have been stuck in spinOnce too long, return error */
216  configured_=false;
217  return -2;
218  }
219  }else if( mode_ == MODE_PROTOCOL_VER ){
220  if(data == PROTOCOL_VER){
221  mode_++;
222  }else{
223  mode_ = MODE_FIRST_FF;
224  if (configured_ == false)
225  requestSyncTime(); /* send a msg back showing our protocol version */
226  }
227  }else if( mode_ == MODE_SIZE_L ){ /* bottom half of message size */
228  bytes_ = data;
229  index_ = 0;
230  mode_++;
231  checksum_ = data; /* first byte for calculating size checksum */
232  }else if( mode_ == MODE_SIZE_H ){ /* top half of message size */
233  bytes_ += data<<8;
234  mode_++;
235  }else if( mode_ == MODE_SIZE_CHECKSUM ){
236  if( (checksum_%256) == 255)
237  mode_++;
238  else
239  mode_ = MODE_FIRST_FF; /* Abandon the frame if the msg len is wrong */
240  }else if( mode_ == MODE_TOPIC_L ){ /* bottom half of topic id */
241  topic_ = data;
242  mode_++;
243  checksum_ = data; /* first byte included in checksum */
244  }else if( mode_ == MODE_TOPIC_H ){ /* top half of topic id */
245  topic_ += data<<8;
246  mode_ = MODE_MESSAGE;
247  if(bytes_ == 0)
248  mode_ = MODE_MSG_CHECKSUM;
249  }else if( mode_ == MODE_MSG_CHECKSUM ){ /* do checksum */
250  mode_ = MODE_FIRST_FF;
251  if( (checksum_%256) == 255){
252  if(topic_ == TopicInfo::ID_PUBLISHER){
253  requestSyncTime();
254  negotiateTopics();
255  last_sync_time = c_time;
256  last_sync_receive_time = c_time;
257  return -1;
258  }else if(topic_ == TopicInfo::ID_TIME){
259  syncTime(message_in);
260  }else if (topic_ == TopicInfo::ID_PARAMETER_REQUEST){
261  req_param_resp.deserialize(message_in);
262  param_recieved= true;
263  }else if(topic_ == TopicInfo::ID_TX_STOP){
264  configured_ = false;
265  }else{
266  if(subscribers[topic_-100])
267  subscribers[topic_-100]->callback( message_in );
268  }
269  }
270  }
271  }
272 
273  /* occasionally sync time */
274  if( configured_ && ((c_time-last_sync_time) > (SYNC_SECONDS*500) )){
275  requestSyncTime();
276  last_sync_time = c_time;
277  }
278 
279  return 0;
280  }

419  virtual int publish(int id, const Msg * msg)
420  {
421  if(id >= 100 && !configured_)
422  return 0;
423 
424  /* serialize message */
425  unsigned int l = msg->serialize(message_out+7);
426 
427  /* setup the header */
428  message_out[0] = 0xff;
429  message_out[1] = PROTOCOL_VER;
430  message_out[2] = (unsigned char) ((unsigned int)l&255);
431  message_out[3] = (unsigned char) ((unsigned int)l>>8);
432  message_out[4] = 255 - ((message_out[2] + message_out[3])%256);
433  message_out[5] = (unsigned char) ((int)id&255);
434  message_out[6] = (unsigned char) ((int)id>>8);
435 
436  /* calculate checksum */
437  int chk = 0;
438  for(int i =5; i<l+7; i++)
439  chk += message_out[i];
440  l += 7;
441  message_out[l++] = 255 - (chk%256);
442 
443  if( l <= OUTPUT_SIZE ){
444  hardware_.write(message_out, l);
445  return l;
446  }else{
447  logerror("Message from device dropped: message larger than buffer.");
448  return -1;
449  }
450  }
