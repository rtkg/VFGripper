59 static inline uint32_t mapResolution(uint32_t value, uint32_t from, uint32_t to)
60 {
61  if (from == to)
62  return value;
63  if (from > to)
64  return value >> (from-to);
65  else
66  return value << (to-from);

73 void pwm_setup( uint32_t pwm_pin, uint32_t pwm_freq, int iclock )
74 {
75  uint32_t pwm_duty = 0;
76  uint32_t chan = g_APinDescription[pwm_pin].ulPWMChannel;
77 
78  // SET CLOCK FREQUENCY
79  //------------------------
80  if (iclock==1) pwm_set_clockA_freq( pwm_max_duty_Ncount*pwm_freq );
81  if (iclock==2) pwm_set_clockB_freq( pwm_max_duty_Ncount*pwm_freq );
82 
83 
84  if (pwm_pin>=6 && pwm_pin<=9)
85  {
86  // SET PWM RESOLUTION
87  //------------------------
88  pwm_duty = mapResolution( pwm_duty, pwm_resolution_nbit, PWM_RESOLUTION);
89 
90  // PWM STARTUP AND SETUP CLOCK
91  //-------------------------------
92  pmc_enable_periph_clk( PWM_INTERFACE_ID );
93  PWMC_ConfigureClocks( pwm_clockA_freq, pwm_clockB_freq, VARIANT_MCK );
94 
95  // SETUP PWM FOR pwm_pin
96  //------------------------
97  PIO_Configure( g_APinDescription[pwm_pin].pPort, g_APinDescription[pwm_pin].ulPinType, g_APinDescription[pwm_pin].ulPin, g_APinDescription[pwm_pin].ulPinConfiguration);
98  if (iclock==1) PWMC_ConfigureChannel(PWM_INTERFACE, chan, PWM_CMR_CPRE_CLKA, 0, 0);
99  if (iclock==2) PWMC_ConfigureChannel(PWM_INTERFACE, chan, PWM_CMR_CPRE_CLKB, 0, 0);
100  PWMC_SetPeriod(PWM_INTERFACE, chan, pwm_max_duty_Ncount);
101  PWMC_SetDutyCycle(PWM_INTERFACE, chan, pwm_duty);
102  PWMC_EnableChannel(PWM_INTERFACE, chan);
103  }
